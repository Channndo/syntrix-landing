// For password + enterprise sign-ups as TWO spreadsheets in Drive, use google-apps-script-standalone.txt
// (createConsumerSpreadsheet + createEnterpriseSpreadsheet).
// For auto-created single-sheet workbook see legacy flow below.
//
// 1) Open your Google Sheet → Extensions → Apps Script
// 2) Paste CODE below, set NOTIFY_EMAIL / EXPECTED_SECRET
// 3) Deploy → Web app → Execute as: Me, Who has access: Anyone
// 4) Store deployment URL / token only in Netlify env (never in git or public HTML)
//
// Troubleshooting: EXPECTED_SECRET must be ONE line (no line break inside the quotes) or you get a syntax error.
// Use straight ASCII quotes '...' only — not curly/smart quotes from Word or rich text.
//
// For a completely fresh tab: add a new sheet tab, name it Signups, delete the old Signups tab,
// then run appendHeadersToActiveSignupsSheet() once from the editor (select function → Run).

// ----- begin CODE -----
const NOTIFY_EMAIL = 'your-inbox@example.com';
const SHEET_NAME = 'Signups';

const EXPECTED_SECRET = '';

/** 11 columns — must match appendRow in doPost */
const SIGNUPS_HEADERS = [
  'Timestamp',
  'Email',
  'First name',
  'Last name',
  'Full name',
  'Phone',
  'Business address',
  'Referral',
  'Source',
  'Type',
  'Submitted ISO',
];

/** Old sheets had only "Full name" in column C — inserts First + Last columns. */
function upgradeLegacySignupsLayoutIfNeeded(sh) {
  var c1 = String(sh.getRange(1, 3).getValue()).trim();
  var d1 = String(sh.getRange(1, 4).getValue()).trim();
  if (c1 === 'Full name' && d1 !== 'Last name') {
    sh.insertColumnsBefore(3, 2);
    sh.getRange(1, 3, 1, 5).setValues([['First name', 'Last name', 'Full name']]);
  }
}

/** Run once on an empty Signups tab if you need the header row only */
function appendHeadersToActiveSignupsSheet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) {
    throw new Error('Create a tab named ' + SHEET_NAME + ' first');
  }
  sh.getRange(1, 1, 1, SIGNUPS_HEADERS.length).setValues([SIGNUPS_HEADERS]);
}

function doPost(e) {
  let data;
  try {
    data = JSON.parse(e.postData.contents);
  } catch (err) {
    return jsonOut({ ok: false, error: 'bad_json' });
  }

  if (EXPECTED_SECRET && data.secret !== EXPECTED_SECRET) {
    return jsonOut({ ok: false, error: 'unauthorized' });
  }

  var email = String(data.email || '').trim();
  if (!email) {
    return jsonOut({ ok: false, error: 'no_email' });
  }

  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) {
    sh = ss.insertSheet(SHEET_NAME);
    sh.getRange(1, 1, 1, SIGNUPS_HEADERS.length).setValues([SIGNUPS_HEADERS]);
  } else {
    upgradeLegacySignupsLayoutIfNeeded(sh);
  }

  var ts = new Date();
  var first = String(data.first_name || '').trim();
  var last = String(data.last_name || '').trim();
  var full = String(data.name || '').trim();
  if (!full && (first || last)) {
    full = (first + ' ' + last).trim();
  }
  sh.appendRow([
    ts,
    email,
    first,
    last,
    full,
    data.phone || '',
    data.business_address || '',
    data.referral_source || '',
    data.source || '',
    data.type || '',
    data.ts || '',
  ]);

  MailApp.sendEmail({
    to: NOTIFY_EMAIL,
    subject: '[Syntrix signup] ' + email,
    body:
      'New signup / waitlist entry\n\n' +
      'Email: ' +
      email +
      '\n' +
      'First name: ' +
      first +
      '\n' +
      'Last name: ' +
      last +
      '\n' +
      'Full name: ' +
      (full || '(none)') +
      '\n' +
      'Phone: ' +
      (data.phone || '(none)') +
      '\n' +
      'Business address: ' +
      (data.business_address || '(none)') +
      '\n' +
      'Referral: ' +
      (data.referral_source || '(none)') +
      '\n' +
      'Source: ' +
      (data.source || '') +
      '\n' +
      'Type: ' +
      (data.type || '') +
      '\n' +
      'Submitted (ISO): ' +
      (data.ts || '') +
      '\n' +
      'Received: ' +
      ts.toISOString(),
  });

  return jsonOut({ ok: true });
}

function doGet(e) {
  return ContentService.createTextOutput('Syntrix waitlist webhook - POST JSON here').setMimeType(
    ContentService.MimeType.TEXT
  );
}

function jsonOut(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(
    ContentService.MimeType.JSON
  );
}
