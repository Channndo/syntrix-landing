// =============================================================================
// SYNTRIX — Standalone Apps Script (paste ONLY into Code.gs)
//
// TWO spreadsheets: consumer (password sign-ups) vs enterprise — separate files in Drive.
//
// 1) Set NOTIFY_EMAIL below. Set EXPECTED_SECRET to match Netlify APPS_SCRIPT_SECRET (same string).
// 2) Run createConsumerSpreadsheet() then createEnterpriseSpreadsheet() → copy both URLs from logs.
// 3) Already have sheets but ugly headers? Run repairConsumerSheetFormatting() / repairEnterpriseSheetFormatting().
// 4) Do not Run doPost from the toolbar — no HTTP body. Use testMailFromEditor() for a mail test.
// 5) Deploy → Web app → Anyone — put URL(s) in Netlify:
//    - Same script, two deployments: consumer URL → APPS_SCRIPT_CONSUMER_URL,
//      enterprise URL → APPS_SCRIPT_ENTERPRISE_URL
//    - OR one URL → APPS_SCRIPT_URL (doPost routes by JSON type).
// =============================================================================

function testMailFromEditor() {
  notifyEmail(
    'Syntrix — Apps Script mail test',
    'If you see this message, MailApp and NOTIFY_EMAIL work. Safe to delete.'
  );
  Logger.log('Sent test email to ' + NOTIFY_EMAIL);
}

var PROP_CONSUMER = 'SYNTRIX_CONSUMER_SHEET_ID';
var PROP_ENTERPRISE = 'SYNTRIX_ENTERPRISE_SHEET_ID';

var NOTIFY_EMAIL = 'chandler.hill.24@gmail.com';
var SHEET_CONSUMER = 'Password signups';
var SHEET_ENTERPRISE = 'Enterprise signups';

// Paste the exact secret Netlify sends as JSON field "secret" (must match APPS_SCRIPT_SECRET).
var EXPECTED_SECRET = '';

var CONSUMER_HEADERS = [
  'Timestamp',
  'Email',
  'First name',
  'Last name',
  'Referral',
  'Source',
  'Type',
  'Submitted ISO',
];

var ENTERPRISE_HEADERS = [
  'Timestamp',
  'Business name',
  'Business address',
  'Referral (organization)',
  'Supervisor name',
  'Employee first name',
  'Employee last name',
  'Company email',
  'Phone',
  'Referral (employee)',
  'Source',
  'Type',
  'Submitted ISO',
];

// Pixel widths — readable without clipping emails / ISO strings.
var CONSUMER_COL_WIDTHS = [176, 288, 112, 112, 168, 112, 156, 244];
var ENTERPRISE_COL_WIDTHS = [176, 200, 260, 168, 168, 132, 132, 240, 132, 168, 112, 156, 244];

function applyConsumerSheetFormatting(sh) {
  var n = CONSUMER_HEADERS.length;
  var header = sh.getRange(1, 1, 1, n);
  header.setValues([CONSUMER_HEADERS]);
  header.setFontWeight('bold');
  header.setBackground('#132238');
  header.setFontColor('#e8eef7');
  header.setHorizontalAlignment('center');
  header.setVerticalAlignment('middle');
  header.setWrap(false);

  sh.setFrozenRows(1);
  var i;
  for (i = 0; i < n; i++) {
    sh.setColumnWidth(i + 1, CONSUMER_COL_WIDTHS[i]);
  }

  var maxRows = sh.getMaxRows();
  if (maxRows > 1) {
    var body = sh.getRange(2, 1, maxRows, n);
    body.setVerticalAlignment('top');
    body.setWrap(true);
    sh.getRange(2, 1, maxRows, 1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
    sh.getRange(2, 2, maxRows, 2).setHorizontalAlignment('left');
    sh.getRange(2, 8, maxRows, 8).setHorizontalAlignment('left');
  }
}

function applyEnterpriseSheetFormatting(sh) {
  var n = ENTERPRISE_HEADERS.length;
  var header = sh.getRange(1, 1, 1, n);
  header.setValues([ENTERPRISE_HEADERS]);
  header.setFontWeight('bold');
  header.setBackground('#132238');
  header.setFontColor('#e8eef7');
  header.setHorizontalAlignment('center');
  header.setVerticalAlignment('middle');
  header.setWrap(false);

  sh.setFrozenRows(1);
  var i;
  for (i = 0; i < ENTERPRISE_COL_WIDTHS.length && i < n; i++) {
    sh.setColumnWidth(i + 1, ENTERPRISE_COL_WIDTHS[i]);
  }

  var maxRows = sh.getMaxRows();
  if (maxRows > 1) {
    var body = sh.getRange(2, 1, maxRows, n);
    body.setVerticalAlignment('top');
    body.setWrap(true);
    sh.getRange(2, 1, maxRows, 1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
    sh.getRange(2, 3, maxRows, 3).setHorizontalAlignment('left');
    sh.getRange(2, 8, maxRows, 8).setHorizontalAlignment('left');
    sh.getRange(2, 13, maxRows, 13).setHorizontalAlignment('left');
  }
}

/** Run from editor if row 1 was overwritten or columns look cramped — uses Script Properties sheet ID. */
function repairConsumerSheetFormatting() {
  var ss = openConsumerSpreadsheet();
  var sh = ss.getSheetByName(SHEET_CONSUMER);
  if (!sh) throw new Error('Sheet "' + SHEET_CONSUMER + '" not found.');
  applyConsumerSheetFormatting(sh);
  SpreadsheetApp.flush();
  Logger.log('Consumer sheet formatted: ' + ss.getUrl());
}

/** Same for enterprise workbook. */
function repairEnterpriseSheetFormatting() {
  var ss = openEnterpriseSpreadsheet();
  var sh = ss.getSheetByName(SHEET_ENTERPRISE);
  if (!sh) throw new Error('Sheet "' + SHEET_ENTERPRISE + '" not found.');
  applyEnterpriseSheetFormatting(sh);
  SpreadsheetApp.flush();
  Logger.log('Enterprise sheet formatted: ' + ss.getUrl());
}

function createConsumerSpreadsheet() {
  var ss = SpreadsheetApp.create('Syntrix — password signups');
  PropertiesService.getScriptProperties().setProperty(PROP_CONSUMER, ss.getId());
  var sh = ss.getSheets()[0];
  sh.setName(SHEET_CONSUMER);
  applyConsumerSheetFormatting(sh);
  Logger.log('CONSUMER spreadsheet: ' + ss.getUrl());
}

function createEnterpriseSpreadsheet() {
  var ss = SpreadsheetApp.create('Syntrix — enterprise signups');
  PropertiesService.getScriptProperties().setProperty(PROP_ENTERPRISE, ss.getId());
  var sh = ss.getSheets()[0];
  sh.setName(SHEET_ENTERPRISE);
  applyEnterpriseSheetFormatting(sh);
  Logger.log('ENTERPRISE spreadsheet: ' + ss.getUrl());
}

function openConsumerSpreadsheet() {
  var id = PropertiesService.getScriptProperties().getProperty(PROP_CONSUMER);
  if (!id) throw new Error('Run createConsumerSpreadsheet() first.');
  return SpreadsheetApp.openById(id);
}

function openEnterpriseSpreadsheet() {
  var id = PropertiesService.getScriptProperties().getProperty(PROP_ENTERPRISE);
  if (!id) throw new Error('Run createEnterpriseSpreadsheet() first.');
  return SpreadsheetApp.openById(id);
}

function ensureConsumerHeaders(sh) {
  var a1 = String(sh.getRange(1, 1).getDisplayValue()).trim();
  if (a1 !== 'Timestamp') {
    sh.getRange(1, 1, 1, CONSUMER_HEADERS.length).setValues([CONSUMER_HEADERS]);
  }
}

function ensureEnterpriseHeaders(sh) {
  var a1 = String(sh.getRange(1, 1).getDisplayValue()).trim();
  if (a1 !== 'Timestamp') {
    sh.getRange(1, 1, 1, ENTERPRISE_HEADERS.length).setValues([ENTERPRISE_HEADERS]);
  }
}

function appendConsumerRow(data) {
  var ss = openConsumerSpreadsheet();
  var sh = ss.getSheetByName(SHEET_CONSUMER);
  if (!sh) {
    sh = ss.insertSheet(SHEET_CONSUMER);
  }
  ensureConsumerHeaders(sh);
  applyConsumerSheetFormatting(sh);

  var ts = new Date();
  var iso = String(data.ts || '');
  sh.appendRow([
    ts,
    String(data.email || ''),
    String(data.first_name || ''),
    String(data.last_name || ''),
    String(data.referral_source || ''),
    String(data.source || ''),
    String(data.type || ''),
    iso,
  ]);

  var lr = sh.getLastRow();
  var n = CONSUMER_HEADERS.length;
  sh.getRange(lr, 1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
  sh.getRange(lr, 1, lr, n).setVerticalAlignment('top');

  return ts;
}

function appendEnterpriseRow(data) {
  var ss = openEnterpriseSpreadsheet();
  var sh = ss.getSheetByName(SHEET_ENTERPRISE);
  if (!sh) {
    sh = ss.insertSheet(SHEET_ENTERPRISE);
  }
  ensureEnterpriseHeaders(sh);
  applyEnterpriseSheetFormatting(sh);

  var ts = new Date();
  var iso = String(data.ts || '');
  sh.appendRow([
    ts,
    String(data.business_name || ''),
    String(data.business_address || ''),
    String(data.referral_source || ''),
    String(data.supervisor_name || ''),
    String(data.employee_first_name || ''),
    String(data.employee_last_name || ''),
    String(data.email || data.company_email || ''),
    String(data.phone || ''),
    String(data.referral_source_secondary || ''),
    String(data.source || ''),
    String(data.type || ''),
    iso,
  ]);

  var lr = sh.getLastRow();
  var n = ENTERPRISE_HEADERS.length;
  sh.getRange(lr, 1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
  sh.getRange(lr, 1, lr, n).setVerticalAlignment('top');

  return ts;
}

function notifyEmail(subject, body) {
  var subj = subject != null && String(subject).trim() !== '' ? String(subject) : '(Syntrix notify)';
  var txt = body != null ? String(body) : '';
  MailApp.sendEmail({
    to: NOTIFY_EMAIL,
    subject: subj,
    body: txt,
  });
}

function doPost(e) {
  if (!e || !e.postData || e.postData.contents == null || e.postData.contents === '') {
    return jsonOut({
      ok: false,
      error: 'no_http_post',
      hint:
        'doPost is only called when Netlify POSTs to the web app URL. From the editor, run createConsumerSpreadsheet / testMailFromEditor instead.',
    });
  }
  var data;
  try {
    data = JSON.parse(e.postData.contents);
  } catch (err) {
    return jsonOut({ ok: false, error: 'bad_json' });
  }

  if (EXPECTED_SECRET && data.secret !== EXPECTED_SECRET) {
    return jsonOut({ ok: false, error: 'unauthorized' });
  }

  var typ = String(data.type || '');
  try {
    if (typ === 'enterprise_registration') {
      var em = String(data.email || data.company_email || '').trim();
      if (!em) return jsonOut({ ok: false, error: 'no_email' });
      var tsE = appendEnterpriseRow(data);
      notifyEmail(
        '[Syntrix enterprise] ' + em,
        formatEnterpriseEmail(data, tsE)
      );
    } else {
      var emC = String(data.email || '').trim();
      if (!emC) return jsonOut({ ok: false, error: 'no_email' });
      var tsC = appendConsumerRow(data);
      notifyEmail(
        '[Syntrix signup] ' + emC,
        formatConsumerEmail(data, tsC)
      );
    }
  } catch (err) {
    return jsonOut({ ok: false, error: 'sheet_error', message: String(err && err.message ? err.message : err) });
  }

  return jsonOut({ ok: true });
}

    '\n' +
    'Name: ' +

}

function doGet(e) {
  return ContentService.createTextOutput(
    'Syntrix webhook — POST JSON. Run createConsumerSpreadsheet() and createEnterpriseSpreadsheet() once.'
  ).setMimeType(ContentService.MimeType.TEXT);
}

function jsonOut(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
